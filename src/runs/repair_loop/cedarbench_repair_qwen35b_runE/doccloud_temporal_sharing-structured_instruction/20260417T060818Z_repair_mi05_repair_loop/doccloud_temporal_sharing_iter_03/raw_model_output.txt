<cedar_policy>
permit (
    principal is Public,
    action == Action::"ViewDocument",
    resource is Document
) when {
    resource.publicAccess == "view" &&
    context.requestTime <= resource.shareExpiry
};

permit (
    principal is User,
    action == Action::"ViewDocument",
    resource is Document
) when {
    principal == resource.owner ||
    principal in resource.viewACL ||
    (resource.publicAccess == "edit" && context.requestTime <= resource.shareExpiry)
};

permit (
    principal is User,
    action == Action::"ModifyDocument",
    resource is Document
) when {
    principal == resource.owner ||
    principal in resource.modifyACL ||
    (resource.publicAccess == "edit" && context.requestTime <= resource.shareExpiry)
};

permit (
    principal is User,
    action == Action::"DeleteDocument",
    resource is Document
) when {
    principal == resource.owner
};

permit (
    principal is User,
    action == Action::"AddToShareACL",
    resource is Document
) when {
    principal == resource.owner ||
    principal in resource.manageACL
};

permit (
    principal is User,
    action == Action::"EditIsPrivate",
    resource is Document
) when {
    principal == resource.owner ||
    principal in resource.manageACL
};

permit (
    principal is User,
    action == Action::"EditPublicAccess",
    resource is Document
) when {
    principal == resource.owner ||
    principal in resource.manageACL
};

permit (
    principal is User,
    action == Action::"CreateDocument",
    resource is Drive
);

permit (
    principal is User,
    action == Action::"CreateGroup",
    resource is Drive
);

permit (
    principal is User,
    action == Action::"ModifyGroup",
    resource is Group
) when {
    principal == resource.owner
};

permit (
    principal is User,
    action == Action::"DeleteGroup",
    resource is Group
) when {
    principal == resource.owner
};

forbid (
    principal is User,
    action in [Action::"ViewDocument", Action::"ModifyDocument"],
    resource is Document
) when {
    principal in resource.owner.blocked ||
    resource.owner in principal.blocked
};

forbid (
    principal is User,
    action in [
        Action::"ViewDocument",
        Action::"ModifyDocument",
        Action::"AddToShareACL",
        Action::"EditIsPrivate",
        Action::"EditPublicAccess"
    ],
    resource is Document
) when {
    context.requestTime > resource.shareExpiry
} unless {
    principal == resource.owner
};

forbid (
    principal is User,
    action == Action::"ModifyDocument",
    resource is Document
) when {
    resource.isLocked
};

forbid (
    principal is User,
    action in [
        Action::"ViewDocument",
        Action::"ModifyDocument",
        Action::"CreateDocument",
        Action::"CreateGroup",
        Action::"AddToShareACL",
        Action::"DeleteDocument",
        Action::"EditIsPrivate",
        Action::"EditPublicAccess",
        Action::"ModifyGroup",
        Action::"DeleteGroup"
    ],
    resource
) when {
    !context.is_authenticated
};
</cedar_policy>
