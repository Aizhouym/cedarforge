<cedar_policy>
permit (
    principal is Subscriber,
    action == Action::"watch",
    resource is Movie
) when {
    !resource.isFree &&
    !resource.needsRentOrBuy &&
    (
        !resource.allowedRegions.contains(context.region) ||
        (
            resource.rating == "G" ||
            resource.rating == "PG"
        ) &&
        (
            context.now.datetime.offset(context.now.localTimeOffset).hour >= 6 &&
            context.now.datetime.offset(context.now.localTimeOffset).hour < 21
        )
    )
};

permit (
    principal is Subscriber,
    action == Action::"watch",
    resource is Show
) when {
    !resource.isFree &&
    (
        !resource.isEarlyAccess ||
        context.now.datetime >= resource.releaseDate.offset(duration("-24h"))
    ) &&
    (
        !resource.allowedRegions.contains(context.region) ||
        (
            resource.rating == "G" ||
            resource.rating == "PG"
        ) &&
        (
            context.now.datetime.offset(context.now.localTimeOffset).hour >= 6 &&
            context.now.datetime.offset(context.now.localTimeOffset).hour < 21
        )
    )
};

permit (
    principal is FreeMember,
    action == Action::"watch",
    resource is Movie
) when {
    resource.isFree &&
    !resource.allowedRegions.contains(context.region)
};

permit (
    principal is FreeMember,
    action == Action::"watch",
    resource is Show
) when {
    resource.isFree &&
    !resource.allowedRegions.contains(context.region)
};

permit (
    principal is Subscriber,
    action == Action::"rent",
    resource is Movie
) when {
    resource.isOscarNominated &&
    context.now.datetime >= datetime("2025-02-01T00:00:00Z") &&
    context.now.datetime <= datetime("2025-03-31T23:59:59Z")
};

permit (
    principal is Subscriber,
    action == Action::"buy",
    resource is Movie
) when {
    resource.isOscarNominated &&
    context.now.datetime >= datetime("2025-02-01T00:00:00Z") &&
    context.now.datetime <= datetime("2025-03-31T23:59:59Z")
};

permit (
    principal is Subscriber,
    action == Action::"download",
    resource is Movie
) when {
    !resource.isFree
};

permit (
    principal is Subscriber,
    action == Action::"download",
    resource is Show
) when {
    !resource.isFree
};

forbid (
    principal is Subscriber,
    action == Action::"watch",
    resource is Movie
) when {
    principal.profile.isKid &&
    (
        resource.rating != "G" ||
        resource.rating != "PG"
    )
};

forbid (
    principal is Subscriber,
    action == Action::"watch",
    resource is Show
) when {
    principal.profile.isKid &&
    (
        resource.rating != "G" ||
        resource.rating != "PG"
    )
};

forbid (
    principal is Subscriber,
    action == Action::"watch",
    resource is Movie
) when {

    (
        context.now.datetime.offset(context.now.localTimeOffset).hour < 6 ||

    )
};

forbid (
    principal is Subscriber,
    action == Action::"watch",
    resource is Show
) when {
    principal.profile.isKid &&
    (
        context.now.datetime.offset(context.now.localTimeOffset).hour < 6 ||
        context.now.datetime.offset(context.now.localTimeOffset).hour >= 21
    )
};

forbid (
    principal is FreeMember,
    action == Action::"watch",
    resource is Movie
) when {
    resource.allowedRegions.contains(context.region)
};

forbid (
    principal is FreeMember,
    action == Action::"watch",
    resource is Show
) when {
    resource.allowedRegions.contains(context.region)
};

forbid (

) when {
