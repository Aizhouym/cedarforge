<cedar_policy>
permit (
    principal is User,
    action == Action::"view",
    resource is Secret
) when {
    (resource.sensitivityLevel == 1) ||
    (resource.sensitivityLevel == 2 && context.hasMfa) ||
    (resource.sensitivityLevel == 3 && context.hasMfa && context.hasManagerApproval)
};

permit (
    principal is User,
    action == Action::"rotate",
    resource is Secret
) when {
    (principal.role == "security_officer" || principal.role == "admin") &&
    (context.hasMfa &&
     (resource.sensitivityLevel != 3 || context.hasSecurityReview))
};

permit (
    principal is User,
    action == Action::"revoke",
    resource is Secret
) when {
    principal.role == "admin" &&
    context.hasMfa &&
    context.hasManagerApproval
};
</cedar_policy>
