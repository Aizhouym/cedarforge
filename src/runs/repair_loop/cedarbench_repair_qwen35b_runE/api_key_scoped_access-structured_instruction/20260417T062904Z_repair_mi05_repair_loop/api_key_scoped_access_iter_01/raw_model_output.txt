<cedar_policy>
permit (
    principal is ApiKey,
    action == Action::"readDocument",
    resource is Document
) when {
    principal.organization == resource.organization &&
    principal.scopes.contains("read:document") &&
    context.now.datetime < principal.expiresAt &&
    !principal.revoked
};

permit (
    principal is ApiKey,
    action == Action::"writeDocument",
    resource is Document
) when {
    principal.organization == resource.organization &&
    principal.scopes.contains("write:document") &&
    context.now.datetime < principal.expiresAt &&
    !principal.revoked
};

permit (
    principal is ApiKey,
    action == Action::"deleteDocument",
    resource is Document
) when {
    principal.organization == resource.organization &&
    principal.scopes.contains("delete:document") &&
    context.now.datetime < principal.expiresAt &&
    !principal.revoked
};

permit (
    principal is ApiKey,
    action == Action::"readInvoice",
    resource is Invoice
) when {
    principal.organization == resource.organization &&
    principal.scopes.contains("read:invoice") &&
    context.now.datetime < principal.expiresAt &&
    !principal.revoked
};

permit (
    principal is ApiKey,
    action == Action::"writeInvoice",
    resource is Invoice
) when {
    principal.organization == resource.organization &&
    principal.scopes.contains("write:invoice") &&
    context.now.datetime < principal.expiresAt &&
    !principal.revoked &&
    !resource.submitted
};

forbid (
    principal is ApiKey,
    action in [
        Action::"readDocument",
        Action::"writeDocument",
        Action::"deleteDocument",
        Action::"readInvoice",
        Action::"writeInvoice"
    ],
    resource is Document
) when {
    context.now.datetime >= principal.expiresAt
};

forbid (
    principal is ApiKey,
    action in [
        Action::"readDocument",
        Action::"writeDocument",
        Action::"deleteDocument",
        Action::"readInvoice",
        Action::"writeInvoice"
    ],
    resource is Document
) when {
    principal.revoked
};

forbid (
    principal is ApiKey,
    action == Action::"writeInvoice",
    resource is Invoice
) when {
    resource.submitted
};
</cedar_policy>
