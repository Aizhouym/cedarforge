<cedar_policy>
permit (
    principal is User,
    action == Action::"UpdateWorkspace",
    resource is Workspace
) when {
    principal in Role::"Role-A" &&
    (
        !resource.tags.has("production_status") ||
        !principal.allowedTagsForRole.has("Role-A") ||
        !principal.allowedTagsForRole["Role-A"].has("production_status") ||
        principal.allowedTagsForRole["Role-A"]["Role-A"].production_status.contains("ALL") ||
        resource.tags.production_status.contains("ALL") ||
        principal.allowedTagsForRole["Role-A"]["Role-A"].production_status.containsAll(resource.tags.production_status)
    ) &&
    (
        !resource.tags.has("country") ||
        !principal.allowedTagsForRole.has("Role-A") ||
        !principal.allowedTagsForRole["Role-A"].has("country") ||
        principal.allowedTagsForRole["Role-A"]["Role-A"].country.contains("ALL") ||
        resource.tags.country.contains("ALL") ||
        principal.allowedTagsForRole["Role-A"]["Role-A"].country.containsAll(resource.tags.country)
    ) &&
    (
        !resource.tags.has("stage") ||
        !principal.allowedTagsForRole.has("Role-A") ||
        !principal.allowedTagsForRole["Role-A"].has("stage") ||
        principal.allowedTagsForRole["Role-A"]["Role-A"].stage.contains("ALL") ||
        resource.tags.stage.contains("ALL") ||
        principal.allowedTagsForRole["Role-A"]["Role-A"].stage.containsAll(resource.tags.stage)
    )
};

forbid (
    principal is User,
    action in [Action::"UpdateWorkspace", Action::"DeleteWorkspace", Action::"ReadWorkspace"],
    resource is Workspace
) when {
    principal in Role::"Role-A" &&
    resource.sensitivity > 3
};

forbid (
    principal is User,
    action == Action::"ReadWorkspace",
    resource is Workspace
) when {
    principal in Role::"Role-B" &&
    resource.sensitivity > 1
};
